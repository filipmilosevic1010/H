#include <stdio.h>
#include <stdlib.h>
#pragma pack(4)
#include "fajl.h"

void pretvoriUBigEndian(char *buffer){
    int N = *((int*)buffer);
    int i;
    for(i = 0; i < N; i++){
        char *baza = buffer + sizeof(int) + i * sizeof(fajl);
        int val = *((int*)(baza + 8));
        baza[8]  = (val >> 24) & 0xFF;
        baza[9]  = (val >> 16) & 0xFF;
        baza[10] = (val >>  8) & 0xFF;
        baza[11] =  val        & 0xFF;
    }
}

void ispisiHex(int val){
    char cifre[] = "0123456789ABCDEF";
    char buf[9];
    int i;

    buf[8] = '\0';
    for(i = 7; i >= 0; i--){
        buf[i] = cifre[val & 0xF];
        val >>= 4;
    }

    i = 0;
    while(i < 7 && buf[i] == '0') i++;
    printf("%s", buf + i);
}

void ispisiNiz(char *buffer){
    int N = *((int*)buffer);
    int i;

    printf("Broj fajlova: %d\n", N);

    for(i = 0; i < N; i++){
        char *baza = buffer + sizeof(int) + i * sizeof(fajl);

        short dan    = *((short*)(baza + 0));
        short mesec  = *((short*)(baza + 2));
        short godina = *((short*)(baza + 4));
        char pristup = *((char*) (baza + 6));

        int velicina = (((int)baza[8]  & 0xFF) << 24) |
                       (((int)baza[9]  & 0xFF) << 16) |
                       (((int)baza[10] & 0xFF) <<  8) |
                        ((int)baza[11] & 0xFF);

        char *ime = baza + 12;

        printf("\nFajl %d:\n", i + 1);
        printf("  Datum    : %hd.%hd.%hd\n", dan, mesec, godina);
        printf("  Pristup  : %d\n", pristup);
        printf("  Velicina : ");
        ispisiHex(velicina);
        printf("\n");
        printf("  Ime      : %s\n", ime);
    }
}

int main(){
    char *buffer = napraviNiz();

    pretvoriUBigEndian(buffer);

    ispisiNiz(buffer);

    free(buffer);
    return 0;
}
